function generateArray() {
    var arrayLength = parseInt(document.getElementById('arrayLength').value);

    if (isNaN(arrayLength) || arrayLength <= 0) {
        alert("Будь ласка, введіть дійсне додатнє число для довжини масиву.");
        return;
    }

    document.getElementById('output').innerHTML = '';

    var array = [];
    for (var i = 0; i < arrayLength; i++) {
        array.push(Math.floor(Math.random() * 100) + 1);
    }

    displayArray("Вхідний масив: ", array);

    // Фільтрація парних та непарних значень
    var evenValues = array.filter(function (value) { return value % 2 === 0; });
    var oddValues = array.filter(function (value) { return value % 2 !== 0; });

    // Знаходження максимального та мінімального значення для парних та непарних масивів
    var maxEven = evenValues.length > 0 ? Math.max.apply(null, evenValues) : undefined;
    var minEven = evenValues.length > 0 ? Math.min.apply(null, evenValues) : undefined;

    var maxOdd = oddValues.length > 0 ? Math.max.apply(null, oddValues) : undefined;
    var minOdd = oddValues.length > 0 ? Math.min.apply(null, oddValues) : undefined;

    // Відображення результатів
    displayResults("Парні - Макс: " + (maxEven !== undefined ? maxEven : "немає") + ", Мін: " + (minEven !== undefined ? minEven : "немає"));
    displayResults("Непарні - Макс: " + (maxOdd !== undefined ? maxOdd : "немає") + ", Мін: " + (minOdd !== undefined ? minOdd : "немає"));
    

    // Відсортувати масив у порядку зменшення за допомогою сортування вибору
    selectionSort(array);
    // Функція сортування вибором у порядку зменшення
function selectionSort(array) {
    var n = array.length;

    for (var i = 0; i < n - 1; i++) {
        var maxIndex = i;

        for (var j = i + 1; j < n; j++) {
            if (array[j] > array[maxIndex]) {
                maxIndex = j;
            }
        }

        // Обміняти знайдений максимальний елемент з першим
        var temp = array[maxIndex];
        array[maxIndex] = array[i];
        array[i] = temp;
    }
}


    // Показати відсортований масив
    displayArray("Відсортований масив (у зворотньому порядку): ", array);
}
